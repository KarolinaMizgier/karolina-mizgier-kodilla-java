SELECT I.SUMMARY, IL.NAME
FROM ISSUES I JOIN ISSUELIST IL ON I.ISSUESLIST_ID = IL.ID

SELECT I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT I.USER_ID_ASSIGNEDTO, COUNT(*) AS TASKS_NUMBER, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U ON i.USER_ID_ASSIGNEDTO=U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
